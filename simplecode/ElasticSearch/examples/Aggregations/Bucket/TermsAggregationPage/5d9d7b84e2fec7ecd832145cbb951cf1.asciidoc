var searchResponse = client.Search<object>(s => s
    .AllIndices()
    .Size(0)
    .Aggregations(a => a
        .Terms("expired_sessions", t => t
            .Field("account_id")
            .Include(0, 20)
            .Size(10000)
            .Order(o => o
                .Ascending("last_access")
            )
            .Aggregations(aa => aa
                .Max("last_access", m => m
                    .Field("access_date")
                )
            )
        )
    )
);
