var searchResponse = client.Search<object>(s => s
    .AllIndices()
    .Query(q =>
        q.DisMax(c => c
            .Queries(
                qs => qs.MatchPhrasePrefix(m => m.Field("subject").Query("quick brown f")),
                qs => qs.MatchPhrasePrefix(m => m.Field("message").Query("quick brown f"))
            )
        )
    )
