var searchResponse = client.Search<object>(s => s
    .AllIndices()
    .Query(q =>
        q.DisMax(c => c
            .Queries(
                qs => qs.Match(m => m.Field("subject").Query("brown fox")),
                qs => qs.Match(m => m.Field("message").Query("brown fox"))
            )
            .TieBreaker(0.3)
        )
    )
);
