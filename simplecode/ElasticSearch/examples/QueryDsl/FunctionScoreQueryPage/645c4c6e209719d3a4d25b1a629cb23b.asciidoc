var searchResponse = client.Search<object>(s => s
    .AllIndices()
    .Query(q => q
        .FunctionScore(fs => fs
            .Functions(fun => fun
                .RandomScore(rs => rs
                    .Seed(10)
                    .Field("_seq_no")
                )
            )
        )
    )
);
