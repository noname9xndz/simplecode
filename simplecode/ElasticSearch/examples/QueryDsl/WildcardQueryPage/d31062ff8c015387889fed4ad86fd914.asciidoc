var searchResponse = client.Search<object>(s => s
    .AllIndices()
    .Query(q => q
        .Wildcard(w => w
            .Field("user")
            .Value("ki*y")
            .Boost(1)
            .Rewrite(MultiTermQueryRewrite.ConstantScore)

    )
