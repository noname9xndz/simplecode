var searchResponse = client.Search<Tweet>(s => s
    .Index("twitter")
    .Query(q => q
        .Match(m => m
            .Field(f => f.Message)
            .Query("elasticsearch")
        )
    )
    .Collapse(c => c
        .Field(f => f.User)
        .InnerHits(ih => ih
            .Name("last_tweets")
            .Size(5)
            .Sort(so => so
                .Ascending("date")
            )
        )
        .MaxConcurrentGroupSearches(4)
    )
    .Sort(so => so
        .Field(f => f.Likes, SortOrder.Descending)
    )
);
