var searchResponse = client.Search<Tweet>(s => s
    .Index("twitter")
    .Query(q => q
        .Match(m => m
            .Field(f => f.Message)
            .Query("elasticsearch")
        )
    )
    .Collapse(c => c
        .Field(f => f.User)
    )
    .Sort(so => so
        .Field(f => f.Likes, SortOrder.Descending)
    )
    .From(10)
);
