var searchResponse = client.Search<Tweet>(s => s
    .Index("twitter")
    .Size(0)
    .FilterPath(new[] { "hits.total" }) //<1>
    .Query(q => q
        .Range(r => r
            .Field(f => f.Likes)
            .LessThan(10)
        )
    )
);
